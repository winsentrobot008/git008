"""
Python program to calculate the sum of numbers from 1 to 100
"""

# Method 1: Using a loop
sum_loop = 0
for i in range(1, 101):
    sum_loop += i
print(f"Method 1 (using loop): 1 + 2 + ... + 100 = {sum_loop}")

# Method 2: Using the mathematical formula n*(n+1)/2
n = 100
sum_formula = n * (n + 1) // 2
print(f"Method 2 (using formula): 1 + 2 + ... + 100 = {sum_formula}")

# Method 3: Using the built-in sum function
sum_builtin = sum(range(1, 101))
print(f"Method 3 (using sum()): 1 + 2 + ... + 100 = {sum_builtin}")

# All three methods should give the same result: 5050
assert sum_loop == sum_formula == sum_builtin == 5050, "All methods should equal 5050"
print("\n✅ All methods confirmed: The sum of 1 to 100 is 5050")
