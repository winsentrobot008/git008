# ==================== 八字计算 ====================
def get_year_gan_zhi(year):
    """获取年柱天干地支"""
    gan_idx = (year - 4) % 10
    zhi_idx = (year - 4) % 12
    return TIAN_GAN[gan_idx], DI_ZHI[zhi_idx]

def get_month_gan_zhi(year, month):
    """获取月柱天干地支（简化版）"""
    # 年干索引
    year_gan_idx = (year - 4) % 10
    # 月干从年干开始推算
    month_gan_start = (year_gan_idx % 5) * 2
    gan_idx = (month_gan_start + month - 1) % 10
    zhi_idx = (month + 1) % 12
    return TIAN_GAN[gan_idx], DI_ZHI[zhi_idx]

def get_day_gan_zhi(year, month, day):
    """获取日柱天干地支（使用简化算法）"""
    # 使用基准日期：1900年1月1日为甲子日
    base_date = date(1900, 1, 1)
    target_date = date(year, month, day)
    delta = (target_date - base_date).days
    gan_idx = delta % 10
    zhi_idx = delta % 12
    return TIAN_GAN[gan_idx], DI_ZHI[zhi_idx]

def get_hour_gan_zhi(day_gan, hour):
    """获取时柱天干地支"""
    day_gan_idx = TIAN_GAN.index(day_gan)
    # 时支索引（每两小时一个时辰）
    zhi_idx = (hour + 1) // 2 % 12
    # 时干根据日干推算
    gan_idx = (day_gan_idx % 5) * 2 + zhi_idx
    gan_idx = gan_idx % 10
    return TIAN_GAN[gan_idx], DI_ZHI[zhi_idx]

def calculate_bazi(year, month, day, hour):
    """计算八字"""
    yg, yz = get_year_gan_zhi(year)
    mg, mz = get_month_gan_zhi(year, month)
    dg, dz = get_day_gan_zhi(year, month, day)
    hg, hz = get_hour_gan_zhi(dg, hour)
    
    bazi = {
        'year': {'gan': yg, 'zhi': yz, 'wuxing_gan': TIAN_GAN_WU_XING[yg], 'wuxing_zhi': DI_ZHI_WU_XING[yz]},
        'month': {'gan': mg, 'zhi': mz, 'wuxing_gan': TIAN_GAN_WU_XING[mg], 'wuxing_zhi': DI_ZHI_WU_XING[mz]},
        'day': {'gan': dg, 'zhi': dz, 'wuxing_gan': TIAN_GAN_WU_XING[dg], 'wuxing_zhi': DI_ZHI_WU_XING[dz]},
        'hour': {'gan': hg, 'zhi': hz, 'wuxing_gan': TIAN_GAN_WU_XING[hg], 'wuxing_zhi': DI_ZHI_WU_XING[hz]}
    }
    return bazi

def get_sheng_xiao(year):
    """获取生肖"""
    idx = (year - 4) % 12
    return SHENG_XIAO[idx]

def analyze_wuxing(bazi):
    """分析五行分布"""
    wuxing_count = {'金': 0, '木': 0, '水': 0, '火': 0, '土': 0}
    for pillar in ['year', 'month', 'day', 'hour']:
        wx_gan = bazi[pillar]['wuxing_gan']
        wx_zhi = bazi[pillar]['wuxing_zhi']
        wuxing_count[wx_gan] = wuxing_count.get(wx_gan, 0) + 1
        wuxing_count[wx_zhi] = wuxing_count.get(wx_zhi, 0) + 1
    
    # 找最多和最少的
    max_wx = max(wuxing_count, key=wuxing_count.get)
    min_wx = min(wuxing_count, key=wuxing_count.get)
    
    return wuxing_count, max_wx, min_wx

def analyze_name(lastname, firstname):
    """分析姓名五格"""
    # 计算笔画
    last_strokes = sum(get_stroke_count(c) for c in lastname)
    first_strokes = sum(get_stroke_count(c) for c in firstname)
    total_strokes = last_strokes + first_strokes
    
    # 天格：姓氏笔画+1
    tian_ge = last_strokes + 1
    # 人格：姓氏最后一字+名字第一字
    ren_ge = get_stroke_count(lastname[-1]) + get_stroke_count(firstname[0]) if firstname else get_stroke_count(lastname[-1])
    # 地格：名字笔画相加
    di_ge = first_strokes + 1 if len(firstname) <= 1 else first_strokes
    # 外格：总格-人格+1
    wai_ge = total_strokes - ren_ge + 1
    # 总格：全部笔画
    zong_ge = total_strokes + 1
    
    result = {
        '天格': {'num': tian_ge, 'luck': GE_SHUI.get(tian_ge, '未知')},
        '人格': {'num': ren_ge, 'luck': GE_SHUI.get(ren_ge, '未知')},
        '地格': {'num': di_ge, 'luck': GE_SHUI.get(di_ge, '未知')},
        '外格': {'num': wai_ge, 'luck': GE_SHUI.get(wai_ge, '未知')},
        '总格': {'num': zong_ge, 'luck': GE_SHUI.get(zong_ge, '未知')}
    }
    
    return result

def get_constellation(month, day):
    """获取星座"""
    for m, d, name, symbol in CONSTELLATIONS:
        if month == m and day >= d:
            return name, symbol
        elif month == m and day < d:
            prev = CONSTELLATIONS[CONSTELLATIONS.index((m, d, name, symbol)) - 1]
            return prev[2], prev[3]
    return '摩羯座', '♑'

# ==================== 每日运势 ====================
def daily_fortune(constellation_name):
    """获取每日运势"""
    base = CONSTELLATION_FORTUNES.get(constellation_name, {})
    today = datetime.now()
    # 基于日期生成随机但稳定的运势
    seed = int(hashlib.md5(f"{constellation_name}{today.strftime('%Y%m%d')}".encode()).hexdigest(), 16)
    random.seed(seed)
    
    luck_score = random.randint(60, 99)
    lucky_number = random.randint(1, 50)
    lucky_color = random.choice(['红色', '金色', '蓝色', '绿色', '紫色', '白色', '黑色', '黄色'])
    
    random.seed()  # 重置随机种子
    
    return {
        'score': luck_score,
        'lucky_number': lucky_number,
        'lucky_color': lucky_color,
        'career': base.get('career', '平稳发展'),
        'love': base.get('love', '顺其自然'),
        'wealth': base.get('wealth', '财运平稳'),
        'health': base.get('health', '身体健康'),
        'advice': generate_advice(luck_score, constellation_name)
    }

# ==================== 综合算命 ====================
def comprehensive_fortune(birth_year, birth_month, birth_day, birth_hour, lastname, firstname):
    """综合算命分析"""
    # 1. 八字分析
    bazi = calculate_bazi(birth_year, birth_month, birth_day, birth_hour)
    wx_count, max_wx, min_wx = analyze_wuxing(bazi)
    sheng_xiao = get_sheng_xiao(birth_year)
    
    # 2. 星座分析
    const_name, const_symbol = get_constellation(birth_month, birth_day)
    fortune = daily_fortune(const_name)
    
    # 3. 姓名分析
    name_analysis = analyze_name(lastname, firstname)
    
    # 4. 综合解读
    bazi_text = f"{bazi['year']['gan']}{bazi['year']['zhi']}年 {bazi['month']['gan']}{bazi['month']['zhi']}月 {bazi['day']['gan']}{bazi['day']['zhi']}日 {bazi['hour']['gan']}{bazi['hour']['zhi']}时"
    
    wx_balance = '五行均衡' if max(wx_count.values()) - min(wx_count.values()) <= 2 else f'{max_wx}偏旺，{min_wx}偏弱'
    
    # 姓名总评
    good_count = sum(1 for v in name_analysis.values() if '吉' in v['luck'])
    name_rating = '大吉' if good_count >= 4 else '吉' if good_count >= 2 else '一般'
    
    result = {
        'personal': {
            'name': lastname + firstname,
            'birth': f'{birth_year}年{birth_month}月{birth_day}日{birth_hour}时',
            'age': datetime.now().year - birth_year,
            'sheng_xiao': sheng_xiao,
            'constellation': const_name,
            'constellation_symbol': const_symbol
        },
        'bazi': {
            'full': bazi_text,
            'details': bazi,
            'wuxing': wx_count,
            'wuxing_analysis': wx_balance
        },
        'name_analysis': {
            'wuge': name_analysis,
            'rating': name_rating
        },
        'daily_fortune': fortune,
        'summary': f'【{lastname}{firstname}】生于{birth_year}年，生肖{sheng_xiao}，星座{const_name} {const_symbol}。\n'
                  f'八字：{bazi_text}\n'
                  f'五行分析：{wx_balance}\n'
                  f'姓名评分：{name_rating}\n'
                  f'今日运势评分：{fortune["score"]}分\n'
                  f'幸运数字：{fortune["lucky_number"]}，幸运色：{fortune["lucky_color"]}\n'
                  f'建议：{fortune["advice"]}'
    }
    
    return result

# ==================== API接口类 ====================
class FortuneAPI:
    """算命API接口"""
    
    @staticmethod
    def get_bazi(year, month, day, hour):
        """获取八字"""
        try:
            bazi = calculate_bazi(year, month, day, hour)
            wx_count, max_wx, min_wx = analyze_wuxing(bazi)
            sx = get_sheng_xiao(year)
            
            bazi_text = f"{bazi['year']['gan']}{bazi['year']['zhi']}年 {bazi['month']['gan']}{bazi['month']['zhi']}月 {bazi['day']['gan']}{bazi['day']['zhi']}日 {bazi['hour']['gan']}{bazi['hour']['zhi']}时"
            
            return {
                'success': True,
                'bazi': bazi,
                'bazi_text': bazi_text,
                'wuxing': wx_count,
                'wuxing_analysis': f'{max_wx}偏旺，{min_wx}偏弱',
                'sheng_xiao': sx
            }
        except Exception as e:
            return {'success': False, 'error': str(e)}
    
    @staticmethod
    def get_name_analysis(lastname, firstname):
        """获取姓名分析"""
        try:
            result = analyze_name(lastname, firstname)
            good_count = sum(1 for v in result.values() if '吉' in v['luck'])
            rating = '大吉' if good_count >= 4 else '吉' if good_count >= 2 else '一般'
            return {
                'success': True,
                'wuge': result,
                'rating': rating
            }
        except Exception as e:
            return {'success': False, 'error': str(e)}
    
    @staticmethod
    def get_daily_fortune(constellation_name):
        """获取每日运势"""
        try:
            fortune = daily_fortune(constellation_name)
            return {'success': True, 'fortune': fortune}
        except Exception as e:
            return {'success': False, 'error': str(e)}
    
    @staticmethod
    def get_comprehensive(birth_year, birth_month, birth_day, birth_hour, lastname, firstname):
        """获取综合算命结果"""
        try:
            result = comprehensive_fortune(birth_year, birth_month, birth_day, birth_hour, lastname, firstname)
            return {'success': True, 'result': result}
        except Exception as e:
            return {'success': False, 'error': str(e)}
