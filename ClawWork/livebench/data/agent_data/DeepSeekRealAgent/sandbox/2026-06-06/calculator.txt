import os
import sys

def show_menu():
    """显示计算器菜单"""
    print("\n" + "=" * 40)
    print("          🧮  Python 计算器")
    print("=" * 40)
    print("  请选择运算：")
    print("    1. 加法 (+)")
    print("    2. 减法 (-)")
    print("    3. 乘法 (*)")
    print("    4. 除法 (/)")
    print("    5. 退出")
    print("=" * 40)

def get_number(prompt):
    """获取用户输入的数字，支持整数和小数"""
    while True:
        try:
            num = float(input(prompt))
            return num
        except ValueError:
            print("❌ 输入无效，请输入一个有效的数字！")

def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

def multiply(a, b):
    return a * b

def divide(a, b):
    if b == 0:
        raise ZeroDivisionError("除数不能为零！")
    return a / b

def main():
    """计算器主函数"""
    while True:
        show_menu()
        choice = input("请输入您的选择 (1-5): ").strip()

        if choice == "5":
            print("\n👋 感谢使用计算器，再见！")
            break

        if choice not in ["1", "2", "3", "4"]:
            print("\n❌ 无效选择，请重新输入 (1-5)!")
            continue

        # 获取运算符号显示
        ops = {"1": "+", "2": "-", "3": "*", "4": "/"}
        op_symbol = ops[choice]

        print(f"\n--- 您选择了 {op_symbol} 运算 ---")

        # 获取两个数字
        num1 = get_number("请输入第一个数字: ")
        num2 = get_number("请输入第二个数字: ")

        # 执行运算
        try:
            if choice == "1":
                result = add(num1, num2)
            elif choice == "2":
                result = subtract(num1, num2)
            elif choice == "3":
                result = multiply(num1, num2)
            elif choice == "4":
                result = divide(num1, num2)

            # 格式化输出（如果是整数则显示为整数）
            if result == int(result):
                result_display = int(result)
            else:
                result_display = result

            print(f"\n✅ 结果: {num1} {op_symbol} {num2} = {result_display}")

        except ZeroDivisionError as e:
            print(f"\n❌ 错误: {e}")
        except Exception as e:
            print(f"\n❌ 发生未知错误: {e}")

        # 询问是否继续
        print("\n" + "-" * 40)
        again = input("是否继续计算？(y/n): ").strip().lower()
        if again != "y":
            print("\n👋 感谢使用计算器，再见！")
            break

# ============================================

def advanced_calculator():
    """
    进阶版计算器 - 支持输入完整表达式
    例如：3 + 5, 10.5 * 2, 100 / 4
    """
    print("\n" + "=" * 50)
    print("    🧮  进阶计算器 (支持表达式输入)")
    print("=" * 50)
    print("  支持的运算: +, -, *, /")
    print("  示例: 3 + 5, 10.5 * 2, 100 / 4")
    print("  输入 'exit' 或 'quit' 退出")
    print("=" * 50)

    while True:
        expr = input("\n请输入表达式: ").strip()

        if expr.lower() in ["exit", "quit", "q"]:
            print("👋 再见！")
            break

        if not expr:
            continue

        # 解析表达式
        parts = expr.split()
        if len(parts) != 3:
            print("❌ 格式错误！请使用 '数字 运算符 数字' 的格式")
            print("   示例: 3 + 5")
            continue

        try:
            num1 = float(parts[0])
            operator = parts[1]
            num2 = float(parts[2])

            if operator == "+":
                result = add(num1, num2)
            elif operator == "-":
                result = subtract(num1, num2)
            elif operator == "*":
                result = multiply(num1, num2)
            elif operator == "/":
                result = divide(num1, num2)
            else:
                print(f"❌ 不支持的运算符: '{operator}'，请使用 +, -, *, /")
                continue

            # 格式化显示
            if result == int(result):
                result_display = int(result)
            else:
                result_display = result

            if num1 == int(num1): num1_display = int(num1)
            else: num1_display = num1
            if num2 == int(num2): num2_display = int(num2)
            else: num2_display = num2

            print(f"✅ {num1_display} {operator} {num2_display} = {result_display}")

        except ZeroDivisionError:
            print("❌ 错误：除数不能为零！")
        except ValueError:
            print("❌ 错误：请输入有效的数字！")
        except Exception as e:
            print(f"❌ 发生错误: {e}")

#  图形界面版计算器 (使用 tkinter)
# ============================================

def launch_gui_calculator():
    """
    启动图形界面计算器 (需要 tkinter 支持)
    """
    try:
        import tkinter as tk
        from tkinter import messagebox
    except ImportError:
        print("❌ tkinter 模块不可用，无法启动图形界面。")
        print("   请使用命令行版本 (main 或 advanced_calculator)")
        return

    class CalculatorGUI:
        def __init__(self, root):
            self.root = root
            self.root.title("Python 计算器")
            self.root.geometry("350x500")
            self.root.resizable(False, False)

            # 变量
            self.expression = ""
            self.result_var = tk.StringVar()
            self.result_var.set("0")

            # 创建界面
            self.create_widgets()

        def create_widgets(self):
            # 显示框
            display_frame = tk.Frame(self.root, height=80)
            display_frame.pack(fill=tk.X, padx=10, pady=10)

            display = tk.Entry(
                display_frame,
                textvariable=self.result_var,
                font=("Arial", 24),
                justify=tk.RIGHT,
                bd=8,
                relief=tk.GROOVE,
                state="readonly"
            )
            display.pack(fill=tk.X, ipady=10)

            # 按钮框架
            button_frame = tk.Frame(self.root)
            button_frame.pack(fill=tk.BOTH, expand=True, padx=10, pady=5)

            # 按钮布局
            buttons = [
                ("7", 1, 0), ("8", 1, 1), ("9", 1, 2), ("/", 1, 3),
                ("4", 2, 0), ("5", 2, 1), ("6", 2, 2), ("*", 2, 3),
                ("1", 3, 0), ("2", 3, 1), ("3", 3, 2), ("-", 3, 3),
                ("0", 4, 0), (".", 4, 1), ("=", 4, 2), ("+", 4, 3),
            ]

            # 清除按钮
            clear_btn = tk.Button(
                button_frame,
                text="C",
                font=("Arial", 14, "bold"),
                bg="#ff6b6b",
                fg="white",
                command=self.clear
            )
            clear_btn.grid(row=0, column=0, columnspan=4, sticky="nsew", padx=2, pady=2, ipady=8)

            # 创建数字和运算符按钮
            for (text, row, col) in buttons:
                if text == "=":
                    btn = tk.Button(
                        button_frame,
                        text=text,
                        font=("Arial", 16, "bold"),
                        bg="#4ecdc4",
                        fg="white",
                        command=lambda t=text: self.calculate()
                    )
                elif text in ["+", "-", "*", "/"]:
                    btn = tk.Button(
                        button_frame,
                        text=text,
                        font=("Arial", 16, "bold"),
                        bg="#ffa502",
                        fg="white",
                        command=lambda t=text: self.append_operator(t)
                    )
                else:
                    btn = tk.Button(
                        button_frame,
                        text=text,
                        font=("Arial", 14),
                        bg="#dfe6e9",
                        command=lambda t=text: self.append_number(t)
                    )

                btn.grid(row=row+1, column=col, sticky="nsew", padx=2, pady=2, ipady=8)

            # 配置网格权重
            for i in range(5):
                button_frame.grid_rowconfigure(i, weight=1)
            for i in range(4):
                button_frame.grid_columnconfigure(i, weight=1)

        def append_number(self, num):
            self.expression += num
            self.result_var.set(self.expression)

        def append_operator(self, op):
            if self.expression and self.expression[-1] not in "+-*/":
                self.expression += op
                self.result_var.set(self.expression)

        def clear(self):
            self.expression = ""
            self.result_var.set("0")

        def calculate(self):
            try:
                # 安全检查：防止 eval 执行危险代码
                allowed_chars = set("0123456789.+-*/()")
                if not all(c in allowed_chars for c in self.expression):
                    raise ValueError("不支持的字符")

                result = eval(self.expression)
                if result == int(result):
                    result = int(result)
                self.result_var.set(str(result))
                self.expression = str(result)
            except ZeroDivisionError:
                messagebox.showerror("错误", "除数不能为零！")
                self.clear()
            except Exception:
                messagebox.showerror("错误", "表达式无效！")
                self.clear()

    # 启动 GUI
    root = tk.Tk()
    app = CalculatorGUI(root)
    root.mainloop()

if __name__ == "__main__":
    print("=" * 50)
    print("     🧮  Python 多功能计算器")

    elif mode == "3":
        launch_gui_calculator()
    else:
        print("无效选择，启动基础模式...")
        main()
